import com.zaxxer.hikari.HikariDataSource;
import io.r2dbc.spi.ConnectionFactory;
import io.r2dbc.spi.ConnectionFactories;
import io.r2dbc.spi.ConnectionFactoryOptions;
import org.springframework.boot.context.properties.EnableConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.r2dbc.core.DatabaseClient;

import javax.sql.DataSource;


@Configuration
@EnableConfigurationProperties(DbProps.class)
public class DataSourceConfig {

  @Bean
  public ConnectionFactory reactiveConnection(DbProps p) {
    return ConnectionFactories.get(
      ConnectionFactoryOptions.parse(p.getReactive().getUrl())
        .mutate()
        .option(ConnectionFactoryOptions.USER, p.getReactive().getUsername())
        .option(ConnectionFactoryOptions.PASSWORD, p.getReactive().getPassword())
        .build()
    );
  }

  @Bean
  public DatabaseClient databaseClient(ConnectionFactory cf) {
    return DatabaseClient.create(cf);
  }

  @Bean
  public DataSource jdbcDataSource(DbProps p) {
    HikariDataSource ds = new HikariDataSource();
    ds.setJdbcUrl(p.getJdbc().getUrl());
    ds.setUsername(p.getJdbc().getUsername());
    ds.setPassword(p.getJdbc().getPassword());
    ds.setMaximumPoolSize(p.getJdbc().getPoolMax());
    ds.setMinimumIdle(p.getJdbc().getPoolMin());
    return ds;
  }

  @Bean
  public JdbcTemplate jdbcTemplate(DataSource ds) {
    return new JdbcTemplate(ds);
  }
}
